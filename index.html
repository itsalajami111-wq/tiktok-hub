function renderCards(videos) { const grid = document.getElementById('cardGrid'); if (!videos.length) { document.getElementById('emptyEl').style.display = 'block'; return; } videos.forEach(v => { const scoreClass = v.score >= 8 ? 'score-high' : v.score >= 6 ? 'score-mid' : 'score-low'; const card = document.createElement('div'); card.className = 'card'; card.innerHTML = `
${v.thumbnail ? `` : '🎬'}
AI ${v.score}/10
@${v.author}
${v.desc || 'No caption'}
👁 ${v.views.toLocaleString()} ❤️ ${v.likes.toLocaleString()} 💬 ${v.comments.toLocaleString()}
📌 Topic: ${v.topic}
👥 Audience: ${v.audience_fit}
💡 Idea: ${v.content_idea}
📝 ${v.summary}
▶ Watch
`; grid.appendChild(card); }); }
